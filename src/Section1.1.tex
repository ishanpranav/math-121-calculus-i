\section{Find}
Find the domain and range of the following functions.
\begin{enumerate}
\item
\[f(x)=\sqrt{4-x^4}.\]

\[f:\left[-\sqrt{2},\sqrt{2}\right]\to[0,2],\quad f(x)=\sqrt{4-x^4}.\]
\item
\[f(x)=\ln(x^2-1)+5.\]

\[f:(-\infty,-1)\cup(1,\infty)\to\mathbb{R},\quad f(x)=\ln(x^2-1)+5.\]
\item
\[f(x)=\frac{x^2-1}{x+1}.\]

\[f:(-\infty,-1)\cup(-1,\infty)\to(-\infty,-2)\cup(-2,\infty).\]
\end{enumerate}
\section{Determine}
\begin{enumerate}
\item
\[f(x)=\frac{x}{x^2+1}.\]
\begin{align*}
    f(-x)
    &=\frac{-x}{(-x)^2+1},\\
    &=-\frac{x}{x^2+1},\\
    &=-f(x).
\end{align*}

$f$ is an odd function.
\item
\[f(x)=\cos{x}+x^2+1.\]
\begin{align*}
    f(-x)
    &=\cos(-x)+(-x)^2+1,\\
    &=\cos{x}+x^2+1,\\
    &=f(x).
\end{align*}

$f$ is an even function.
\end{enumerate}
